api-example.js 
const roaster = require('@roaster/sdk');

// Initialize client
const client = roaster.init({
  apiKey: process.env.ROASTER_API_KEY
});

// Get employee schedules
const schedules = await client.rosters.list({
  startDate: '2025-01-01',
  endDate: '2025-01-31',
  status: 'published'
});

// Export to payroll
const hours = await client.timesheets.export({
  format: 'visma',
  period: '2025-01'
});
